grapher.js
a micro js charting framework
current version: 0.35
3.51KB (1.45KB gzipped)

grapher.js is under 5Kb, and brings you full featured 2D graphs.
grapher.js requires no external libraries, images, or css
grapher.js does not use canvas
grapher.js requires only 1 http call or you could inline the whole thing!
grapher.js works practically everywhere!
grapher.js is FREE!

gallery and usage guide

Bar Graph, Horizontal Style, with Bar stack

 
 
 

usage

<script type="text/javascript"> 
	data = {"1":[100,50,20],"2":98,"3":[75,40],"4":60,"5":[70,36],"6":30}
	options =   {"borderColor":"#000",
    "width":"300px",
    "height":"150px",
    "ymax":800,
    "gridlines":true,
	"yunits":["100","90","80","70","60","50","40","30","20","10"],
	"xunits":["Mon","Tue","Wed","Thu","Fri","Mon","Tue","Wed","Thu","Fri"],
	"style":"h"
    }
	grapher.bar("test",data,options);
</script>
